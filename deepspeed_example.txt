Example training launch command (single GPU):

accelerate launch --num_processes 1 --num_machines 1 \
    --deepspeed_config_file ds_config.json \
    MTVLA/train.py \
    --run_name MTVLA_stage1 \
    --action_head diffusion \
    --lr 1e-4 \
    --batch_size 16 \
    --max_steps 5000 \
    --log_interval 10 \
    --ckpt_interval 2500 \
    --warmup_steps 500 \
    --save_dir /path/to/checkpoints/stage1
